def create_custom_information(informable_type, informable_id, args)
  
  data.nil?
  
  data = @client.post("#{@baseurl}/#{@name}/#{informable_type}/#{informable_id}", args)
  
  if !data.nil? && output
    puts JSON.pretty_generate(data)
  end
  
  return data
end
