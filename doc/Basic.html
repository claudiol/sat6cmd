def showstatus(args, output=false)
  if !args.empty?
    puts "This action doesn't take arguments.  Ignoring #{args}"
  end
  
  data = @client.get("/api/"+@client.api_version+"/status",nil)
  
  if output
    puts JSON.pretty_generate(data)
  end
  
end
