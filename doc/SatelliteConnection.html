def delete(location, json_data)

  response = nil
  puts "delete #{location} #{json_data}"

  response = RestClient::Request.new(
      :method => :delete,
      :url => @url+location,
      :user => @user,
      :verify_ssl => @verifyssl,
      :password => @password,
      :headers => { :accept => :json,
      :content_type => :json},
      :payload => JSON.generate(json_data)
  ).execute

  if !response.nil?
    if !response.empty?
      results = JSON.parse(response.to_str)
      return results
    end
  end
end
