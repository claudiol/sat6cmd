def autoattachsubs(args, output=false)
  data = nil
  
  args = cleanargs(args)
  puts args.inspect
  
  if (args['id'].nil?)
    puts "orgautoattach requires an argument --id of type integer"
    return
  else
    id = args.delete('id')
    data = @client.post("#{@baseurl}/#{@name}/#{id}/autoattach_subscriptions",args)
  end
  
  if !data.nil? && output
    puts JSON.pretty_generate(data)
  end
  
  return data
  
end
