def createkey(args)
  if !args.include? '-o'
    puts "#{@name} requires an argument of -o for organization id and -n for name identifier"
    return
  end

  if !args.include? '-n'
        puts "#{@name} requires both arguments of -o for organization id and -n for name identifier"
        return
  end

  org_id = "#{args['-o']}"
  name = "#{args['-n']}"

  if org_id.nil? || name.nil?
    puts "Org id or Name cannot be nil"
    return
  end
  content = {:organization_id => "#{org_id}", :name => "#{name}" }
  data = @client.post("/katello/api/activation_keys", content) ##"/#{args['-c']}/activation_keys",nil)

  if !data.nil?
    puts JSON.pretty_generate(data)
  end

end
